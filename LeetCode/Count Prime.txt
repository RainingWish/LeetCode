class Solution {
    
    private static boolean isPrime(int a){
        
        if (a <= 1) return false;
        
        for(int i = a/2; i>1; i--){
            if(a%i==0){
                return false;
            }
        }
        return true;
    }
    
    public int countPrimes(int n) {
        
        int count = 0;
        
        for(int i=1;i<n;i++){
            if(isPrime(i)==true) count+=1;
        }
        return count;
    }
}
