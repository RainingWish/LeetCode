public int rob(int[] nums) {
        return rob(nums,nums.length-1);
    }
    
    private int rob(int[] nums,int n){
        if(n < 0){
            return 0;
        }else if(n == 0){
            return nums[0];
        }else{
            int A = rob(nums,n-2) + nums[n];
            int B = rob(nums,n-1);
            return A > B ? A : B;
        }
    }
