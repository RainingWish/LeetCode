class Solution {
    public int mySqrt(int x) {
        
        if(x<0) return -1;
        if(x==0) return 0;
        if(x>=1&&x<=3) return 1;
        
        int ans=2;
        
        for(long i=2;i<x/2;i++){
            
            if(i*i<=x){ 
                ans=(int)i; 
            }
            
            if(i*i>x) break;
            
        }
     
        return ans;
    }
}
